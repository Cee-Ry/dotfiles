#install neovim/nvim
sudo dnf install neovim


# install tools
sudo dnf install python3-neovim ripgrep fd-find git


#run
curl -fLo ~/.local/share/nvim/site/autoload/plug.vim --create-dirs \
https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim


#create conf file
mkdir -p ~/.config/nvim
nano ~/.config/nvim/init.vim


#run nvim
nvim


#inside nvim run: skipp this, do this after all are done
:PlugInstall

:q

:CocInstall coc-clangd coc-python coc-json coc-html coc-tsserver
:CocInstall coc-clangd coc-cmake coc-json


##Part2 below
#make sure this are installed
sudo dnf group install c-development development-tools
sudo dnf install clang clang-tools-extra

#init.vim
set number
set relativenumber
set tabstop=4
set shiftwidth=4
set expandtab
set autoindent
set smartindent
set mouse=a
set clipboard=unnamedplus
syntax on

call plug#begin()

" File explorer
Plug 'preservim/nerdtree'

" Fuzzy finder
Plug 'junegunn/fzf'
Plug 'junegunn/fzf.vim'

" Status line
Plug 'vim-airline/vim-airline'

" Theme
Plug 'morhetz/gruvbox'

" Autocomplete and LSP
Plug 'neoclide/coc.nvim', {'branch': 'release'}

" Better C++ syntax highlighting
Plug 'bfrg/vim-cpp-modern'

" Code formatting
Plug 'rhysd/vim-clang-format'

call plug#end()

colorscheme gruvbox

" Keybindings
nnoremap <C-n> :NERDTreeToggle<CR>
nnoremap <C-p> :Files<CR>

" Format C++ file with clang-format
nnoremap <F5> :ClangFormat<CR>

nnoremap <F9> :w<CR>:!g++ % -o %< && ./%<<CR>



#nano ~/.config/nvim/coc-settings.json
{
  "clangd.path": "clangd",
  "clangd.arguments": ["--background-index"],
  "suggest.enablePreview": true
}






##how to use 
| Action             | Key               |
| ------------------ | ----------------- |
| Open file explorer | `Ctrl + n`        |
| Search files       | `Ctrl + p`        |
| Go to definition   | `gd`              |
| Format code        | `F5`              |
| Rename symbol      | `:CocRename`      |
| Show docs          | `K`               |
| Errors list        | `:CocDiagnostics` |
